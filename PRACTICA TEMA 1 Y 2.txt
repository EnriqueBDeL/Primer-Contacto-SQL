4. SELECT ename, job, hiredate

   ORDER BY hiredate;



5. SELECT ename, deptno
   FROM emp
   WHERE deptno = 10 OR deptno = 30
   ORDER BY ename;



6. SELECT ename "EMPLEADO", sal "SALARIO MENSUAL"
   FROM emp

   FROM emp
   WHERE EXTRACT(YEAR FROM hiredate) = 1982;



8. SELECT ename, job
   FROM emp
   WHERE mgr IS NULL;

   FROM emp
   WHERE comm IS NOT NULL
   ORDER BY sal DESC;
